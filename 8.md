def draw_diamond(n):
    for i in range(1, n + 1):
        print(" " * (n - i), end="")
        print("*" * (2 * i - 1))
    for i in range(n - 1, 0, -1):
        print(" " * (n - i), end="")
        print("*" * (2 * i - 1))

# 設定菱形的半徑（高度的一半）
draw_diamond(5)

data = [1, 3, 5, 7, 9]
target = 7
low, high = 0, len(data)-1
while low <= high:
    mid = (low + high) // 2
    if data[mid] == target:
        print(f"Found at {mid}")
        break
    elif data[mid] < target:
        low = mid + 1
    else:
        high = mid - 1

import random 
hero = {"name": "勇者", "hp": 100, "atk": 10}
inventory = ["回復藥水", "火球卷軸"] 
dragon_hp = 50

# 6. 🧩 函式
def attack(target_hp):
    """計算攻擊傷害"""
    damage = random.randint(5, 15) # 2. 🧮 運算
    return target_hp - damage, damage

# 4. 🔄 迴圈 (遊戲主迴圈)
while dragon_hp > 0 and hero["hp"] > 0:
    print(f"\n🐲 惡龍 HP: {dragon_hp} | ⚔️ {hero['name']} HP: {hero['hp']}")
    action = input("你要做什麼？(1.攻擊 2.逃跑): ")
    if action == "1":
        dragon_hp, dmg = attack(dragon_hp) 
        print(f"💥 你造成了 {dmg} 點傷害！")
    elif action == "2":
        print("🏃‍♂️ 你逃跑了！遊戲結束。")
        break
    else:
        print("❓ 不懂的指令")
    if dragon_hp > 0:
        hero["hp"] -= 8
        print("🔥 惡龍噴火！你受傷了 8 點。")
if dragon_hp <= 0:
    print("🎉 恭喜！你打敗了惡龍！")
